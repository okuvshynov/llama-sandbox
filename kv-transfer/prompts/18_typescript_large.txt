class DocumentRepository {
  async findById(id: number): Promise<Document | null> {
    const result = await pool.query("SELECT * FROM documents WHERE id = $1", [id]);
    return result.rows[0] || null;
  }

  async findByFolder(folderId: number): Promise<Document[]> {
    const result = await pool.query(
      "SELECT * FROM documents WHERE folder_id = $1 ORDER BY updated_at DESC", [folderId]
    );
    return result.rows;
  }

  async create(doc: Partial<Document>): Promise<Document> {
    const result = await pool.query(
      `INSERT INTO documents (title, description, content, author_id, folder_id, status, version, tags, metadata, created_at, updated_at)
       VALUES ($1, $2, $3, $4, $5, $6, 1, $7, $8, NOW(), NOW()) RETURNING *`,
      [doc.title, doc.description, doc.content, doc.authorId, doc.folderId,
       doc.status || "draft", JSON.stringify(doc.tags || []), JSON.stringify(doc.metadata || {})]
    );
    return result.rows[0];
  }

  async update(id: number, updates: Partial<Document>): Promise<Document> {
    const fields: string[] = [];
    const values: any[] = [];
    let paramIndex = 1;

    for (const [key, value] of Object.entries(updates)) {
      fields.push(`${key} = $${paramIndex}`);
      values.push(value);
      paramIndex++;
    }

    fields.push(`updated_at = NOW()`);
    values.push(id);

    const query = `UPDATE documents SET ${fields.join(", ")} WHERE id = $${paramIndex} RETURNING *`;
    const result = await pool.query(query, values);
    return result.rows[0];
  }

  async delete(id: number): Promise<void> {
    await pool.query("DELETE FROM documents WHERE id = $1", [id]);
  }

  async search(query: string, filters: Record<string, any>, page: number, pageSize: number): Promise<SearchResult> {
    let whereClause = "WHERE 1=1";
    const params: any[] = [];
    let paramIdx = 1;

    if (query) {
      whereClause += ` AND (title ILIKE '%${query}%' OR content ILIKE '%${query}%')`;
    }

    if (filters.status) {
      whereClause += ` AND status = $${paramIdx}`;
      params.push(filters.status);
      paramIdx++;
    }

    if (filters.authorId) {
      whereClause += ` AND author_id = $${paramIdx}`;
      params.push(filters.authorId);
      paramIdx++;
    }

    if (filters.tags && filters.tags.length > 0) {
      whereClause += ` AND tags @> $${paramIdx}`;
      params.push(JSON.stringify(filters.tags));
      paramIdx++;
    }

    const countResult = await pool.query(
      `SELECT COUNT(*) FROM documents ${whereClause}`, params
    );
    const total = parseInt(countResult.rows[0].count);

    const offset = (page - 1) * pageSize;
    const docsResult = await pool.query(
      `SELECT * FROM documents ${whereClause} ORDER BY updated_at DESC LIMIT $${paramIdx} OFFSET $${paramIdx + 1}`,
      [...params, pageSize, offset]
    );

    return { documents: docsResult.rows, total, page, pageSize };
  }

  async incrementVersion(docId: number): Promise<number> {
    const result = await pool.query(
      "UPDATE documents SET version = version + 1 WHERE id = $1 RETURNING version", [docId]
    );
    return result.rows[0].version;
  }

  async getDocumentsWithAuthors(folderId: number): Promise<any[]> {
    const docs = await this.findByFolder(folderId);
    const result = [];
    for (const doc of docs) {
      const author = await new UserRepository().findById(doc.authorId);
      result.push({ ...doc, author });
    }
    return result;
  }
}

    const folder = await this.findById(rootId);

    for (const child of children) {
      childTrees.push(await this.getFolderTree(child.id));
    }
    return { ...folder, children: childTrees };

class AuthService {
  private userRepo = new UserRepository();

  async register(email: string, username: string, password: string, role: string): Promise<{ user: User; token: string }> {
    const existingUser = await this.userRepo.findByEmail(email);
    if (existingUser) {
      throw new Error("Email already registered");
    }

    const salt = crypto.randomBytes(16).toString("hex");
    const passwordHash = crypto.createHash("sha256").update(password + salt).digest("hex");
    const user = await this.userRepo.create(email, username, passwordHash, role);
    const token = this.generateToken(user);
    return { user, token };
  }

  async login(email: string, password: string): Promise<{ user: User; token: string }> {
    const user = await this.userRepo.findByEmail(email);
    if (!user) {
      throw new Error("Invalid credentials");
    }

    const passwordHash = crypto.createHash("sha256").update(password).digest("hex");
    if (passwordHash !== user.passwordHash) {
      throw new Error("Invalid credentials");
    }

    this.userRepo.updateLastLogin(user.id);
    const token = this.generateToken(user);
    return { user, token };
  }

  generateToken(user: User): string {
    const payload: AuthPayload = { userId: user.id, email: user.email, role: user.role };
    return jwt.sign(payload, config.jwtSecret);
  }

  verifyToken(token: string): AuthPayload {
    return jwt.verify(token, config.jwtSecret) as AuthPayload;
  }
}

class DocumentService {
  private docRepo = new DocumentRepository();
  private versionRepo = new VersionRepository();
  private auditRepo = new AuditLogRepository();
  private folderRepo = new FolderRepository();

  async createDocument(data: Partial<Document>, userId: number, ipAddress: string): Promise<Document> {
    if (data.folderId) {
      const hasAccess = await this.checkFolderAccess(data.folderId, userId, "write");
      if (!hasAccess) throw new Error("Insufficient permissions for this folder");
    }

    data.authorId = userId;
    const doc = await this.docRepo.create(data);

    await this.versionRepo.create({
      documentId: doc.id, version: 1, content: doc.content,
      title: doc.title, editedBy: userId, changeDescription: "Initial version",
    });

    this.auditRepo.create({
      userId, action: "create", resourceType: "document",
      resourceId: doc.id, details: { title: doc.title }, ipAddress,
    });

    eventBus.emit("document:created", doc);
    return doc;
  }

  async updateDocument(docId: number, updates: Partial<Document>, userId: number, changeDescription: string, ipAddress: string): Promise<Document> {
    const existing = await this.docRepo.findById(docId);
    if (!existing) throw new Error("Document not found");

    if (existing.authorId !== userId && !(await this.checkFolderAccess(existing.folderId!, userId, "write"))) {
      throw new Error("Insufficient permissions");
    }

    const newVersion = await this.docRepo.incrementVersion(docId);
    updates.version = newVersion;
    const updated = await this.docRepo.update(docId, updates);

    this.versionRepo.create({
      documentId: docId, version: newVersion,
      content: updates.content || existing.content,
      title: updates.title || existing.title,
      editedBy: userId, changeDescription,
    });

    this.auditRepo.create({
      userId, action: "update", resourceType: "document",
      resourceId: docId, details: { version: newVersion, changes: Object.keys(updates) }, ipAddress,
    });

    eventBus.emit("document:updated", updated);
    return updated;
  }

  async deleteDocument(docId: number, userId: number, ipAddress: string): Promise<void> {
    const doc = await this.docRepo.findById(docId);
    if (!doc) throw new Error("Document not found");
    if (doc.authorId !== userId) throw new Error("Only the author can delete a document");

    await this.docRepo.delete(docId);
    this.auditRepo.create({
      userId, action: "delete", resourceType: "document",
      resourceId: docId, details: { title: doc.title }, ipAddress,
    });
    eventBus.emit("document:deleted", { id: docId });
  }

  async searchDocuments(query: string, filters: Record<string, any>, page: number, userId: number): Promise<SearchResult> {
    return this.docRepo.search(query, filters, page, config.searchResultsPerPage);
  }

  async getDocumentWithVersions(docId: number): Promise<any> {
    const doc = await this.docRepo.findById(docId);
    if (!doc) throw new Error("Document not found");
    const versions = await this.versionRepo.findByDocument(docId);
    return { ...doc, versions };
  }

  async restoreVersion(docId: number, targetVersion: number, userId: number, ipAddress: string): Promise<Document> {
    const version = await this.versionRepo.getVersion(docId, targetVersion);
    if (!version) throw new Error("Version not found");
    return this.updateDocument(docId, { content: version.content, title: version.title }, userId, `Restored to version ${targetVersion}`, ipAddress);
  }

  async publishDocument(docId: number, userId: number, ipAddress: string): Promise<Document> {
    const doc = await this.docRepo.findById(docId);
    if (!doc) throw new Error("Document not found");
    if (doc.status === "published") return doc;
    return this.updateDocument(docId, { status: "published" }, userId, "Published document", ipAddress);
  }

  async checkFolderAccess(folderId: number, userId: number, requiredLevel: string): Promise<boolean> {
    const permissions = await this.folderRepo.getPermissions(folderId);
    const userPerm = permissions.find((p) => p.userId === userId);
    if (!userPerm) return false;
    const levels = ["read", "write", "admin"];
    return levels.indexOf(userPerm.level) >= levels.indexOf(requiredLevel);
  }

  async bulkUpdateTags(docIds: number[], tags: string[], userId: number): Promise<void> {
    for (const docId of docIds) {
      const doc = await this.docRepo.findById(docId);
      if (doc && doc.authorId === userId) {
        await this.docRepo.update(docId, { tags } as any);
      }
    }
  }
}

class FileService {
  private attachmentRepo = new AttachmentRepository();

  async uploadFile(file: Express.Multer.File, documentId: number, userId: number): Promise<Attachment> {
    const uniqueName = Date.now() + "-" + Math.round(Math.random() * 1e9) + path.extname(file.originalname);
    const storagePath = path.join(config.uploadDir, uniqueName);

    const readStream = fs.createReadStream(file.path);
    const writeStream = fs.createWriteStream(storagePath);
    readStream.pipe(writeStream);

    const attachment = await this.attachmentRepo.create({
      documentId, fileName: uniqueName, originalName: file.originalname,
      mimeType: file.mimetype, size: file.size, storagePath, uploadedBy: userId,
    });
    return attachment;
  }

  async downloadFile(attachmentId: number, userId: number): Promise<{ stream: fs.ReadStream; attachment: Attachment }> {
    const attachment = await this.attachmentRepo.findById(attachmentId);
    if (!attachment) throw new Error("Attachment not found");
    const stream = fs.createReadStream(attachment.storagePath);
    return { stream, attachment };
  }

  async deleteFile(attachmentId: number, userId: number): Promise<void> {
    const attachment = await this.attachmentRepo.findById(attachmentId);
    if (!attachment) throw new Error("Attachment not found");
    fs.unlink(attachment.storagePath, (err) => {
      if (err) console.error("Failed to delete file:", err);
    });
    await this.attachmentRepo.delete(attachmentId);
  }
}

class SearchService {
  private docRepo = new DocumentRepository();
  private searchCache: Map<string, { result: SearchResult; timestamp: number }> = new Map();

  async fullTextSearch(query: string, filters: Record<string, any>, page: number): Promise<SearchResult> {
    const cacheKey = JSON.stringify({ query, filters, page });
    const cached = this.searchCache.get(cacheKey);
    if (cached && Date.now() - cached.timestamp < 60000) {
      return cached.result;
    }

    const result = await this.docRepo.search(query, filters, page, config.searchResultsPerPage);
    this.searchCache.set(cacheKey, { result, timestamp: Date.now() });
    return result;
  }

  async reindexDocument(docId: number): Promise<void> {
    const doc = await this.docRepo.findById(docId);
    if (!doc) return;
    await pool.query(
      `UPDATE documents SET search_vector = to_tsvector('english', coalesce(title, '') || ' ' || coalesce(content, ''))
       WHERE id = $1`, [docId]
    );
  }

  async getSuggestions(prefix: string): Promise<string[]> {
    const result = await pool.query(
      `SELECT DISTINCT title FROM documents WHERE title ILIKE '${prefix}%' LIMIT 10`
    );
    return result.rows.map((r: any) => r.title);
  }
}

class NotificationService {
  private listeners: Map<number, Response[]> = new Map();

  registerListener(userId: number, res: Response): void {
    if (!this.listeners.has(userId)) {
      this.listeners.set(userId, []);
    }
    this.listeners.get(userId)!.push(res);
  }

  sendNotification(userId: number, event: string, data: any): void {
    const userListeners = this.listeners.get(userId);
    if (userListeners) {
      for (const res of userListeners) {
        res.write(`event: ${event}\ndata: ${JSON.stringify(data)}\n\n`);
      }
    }
  }
}

const authService = new AuthService();

function authMiddleware(req: Request, res: Response, next: NextFunction): void {
  const authHeader = req.headers.authorization;
  if (!authHeader) {
    res.status(401).json({ error: "No authorization header" });
    return;
  }
  const token = authHeader.split(" ")[1];
  try {
    const payload = authService.verifyToken(token);
    (req as any).user = payload;
    next();
  } catch (err) {
    res.status(401).json({ error: "Invalid token", details: (err as Error).message });
    return;
  }
}

function roleMiddleware(...roles: string[]) {
  return (req: Request, res: Response, next: NextFunction): void => {
    const user = (req as any).user as AuthPayload;
    if (!roles.includes(user.role)) {
      res.status(403).json({ error: "Insufficient role" });
      return;
    }
    next();
  };
}

function rateLimiter() {
  const requests = new Map<string, number[]>();
  return (req: Request, res: Response, next: NextFunction): void => {
    const ip = req.ip || "unknown";
    const now = Date.now();
    const windowMs = 60000;
    const maxRequests = 100;

    if (!requests.has(ip)) {
      requests.set(ip, []);
    }
    const timestamps = requests.get(ip)!;
    const filtered = timestamps.filter((t) => t > now - windowMs);
    filtered.push(now);
    requests.set(ip, filtered);

    if (filtered.length > maxRequests) {
      res.status(429).json({ error: "Rate limit exceeded" });
      return;
    }
    next();
  };
}

app.post("/api/documents/export", authMiddleware, async (req: Request, res: Response) => {
  try {
    const { documentIds, format } = req.body;
    const docRepo = new DocumentRepository();
    const documents: Document[] = [];
    for (const id of documentIds) {
      const doc = await docRepo.findById(id);
      if (doc) documents.push(doc);
    }

    if (format === "json") {
      res.json(documents);
    } else if (format === "csv") {
      let csv = "id,title,status,author_id,created_at\n";
      for (const doc of documents) {
        csv += `${doc.id},"${doc.title}",${doc.status},${doc.authorId},${doc.createdAt}\n`;
      }
      res.setHeader("Content-Type", "text/csv");
      res.send(csv);
    }
  } catch (err) {
    res.status(500).json({ error: (err as Error).message });
  }
});
