class Address(Base):
    street: str
    city: str
    state: str
    zip_code: str
    country: str = "US"
    apt_number: Optional[str] = None

    def format_full(self) -> str:

        if self.apt_number:

        parts.append(f"{self.city}, {self.state} {self.zip_code}")
        parts.append(self.country)
        return "\n".join(parts)

    def validate(self) -> bool:
