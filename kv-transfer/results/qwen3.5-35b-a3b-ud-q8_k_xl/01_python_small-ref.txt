def find_median(nums):
    sorted_nums = sorted(nums)
    n = len(sorted_nums)
    mid = n // 2
    if n % 2 == 0:
        return (sorted_nums[mid] + sorted_nums[mid - 1]) / 2
    return sorted_nums[mid]

def running_median(stream):
    window = []
    medians = []
    for i, val in enumerate(stream):
        window.append(val)
        if len(window) > 5:
            window.pop(0)
        medians.append(find_median(window))
    return medians
